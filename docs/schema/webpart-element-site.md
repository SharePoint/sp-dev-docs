# WebPart element (Site)

**Applies to:** SharePoint 2016 | SharePoint Foundation 2013 | SharePoint Online | SharePoint Server 2013
  
```XML
<WebPart  
    AllUsers = "Text"  
    Assembly = "Text"  
    Cache = "Text"  
    Class = "Text"  
    DisplayName = "Text"  
    Flags = "Text"  
    FrameState = "Integer"  
    IsIncluded = "TRUE" | "FALSE"  
    List = "Text"  
    PerUser = "Text"  
    SolutionId = "Text"  
    Src = "Text"  
    Type = "Integer"  
    Url = "Text"  
    Version = "Text" 
    WebPartIdProperty = "Text"  
    WPTypeId = "Text">
</WebPart>
```
