The following query returns cases where the field base type is 1, the ID
field equals the Project column, and the Template column is not empty.
Results are returned in ascending order by Title and then by ID.

<span codelanguage="xmlLang"></span>
XML 
<span class="copyCode" onclick="CopyCode(this)"
onkeypress="CopyCode_CheckKey(this, event)"
onmouseover="ChangeCopyCodeIcon(this)"
onmouseout="ChangeCopyCodeIcon(this)" tabindex="0">![Copy
code] "Copy code")Copy code</span>
    <Query>
      <Where>
        <And>
          <And>
            <Eq>
              <FieldRef Name="BaseType"></FieldRef>
              <Value Type="Integer">1</Value>
            </Eq>
            <Eq>
              <FieldRef Name="Project"></FieldRef>
              <Value Type="Integer">
                <ProjectProperty Select="ID"></ProjectProperty>
              </Value>
            </Eq>
          </And>
          <IsNotNull>
            <FieldRef Name="Template"></FieldRef>
          </IsNotNull>
        </And>
      </Where>
      <OrderBy>
        <FieldRef Name="Title"></FieldRef>
        <FieldRef Name="ID"></FieldRef>
      </OrderBy>
    </Query>
